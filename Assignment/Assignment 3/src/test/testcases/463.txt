
        calculateSumIterative: function integer(n: integer) {
            sum: integer = 0;
            for (i = 1, i <= n, i+1) {
                sum = sum + i;
            }
            return sum;
        }
        calculateSumRecursive: function integer(n: integer) {
            if (n == 1) {
                return 1;
            }
            return n + calculateSumRecursive(n - 1);
        }
