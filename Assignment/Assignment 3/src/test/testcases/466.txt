
        findMin: function integer(n: integer, arr: array[100] of integer) {
            min: integer = arr[0];
            for (i = 1, i < n, i+1) {
                if (arr[i] < min) {
                    min = arr[i];
                }
            }
            return min;
        }
        main: function void() {
            arr: array [5] of integer = {1, 100, 1000, 0, -1000};
            printInteger(findMin(5, arr));
        }
