
        isPalindrome: function boolean(strs: array[100] of integer, strSize: integer) {
          for (i = 0, i < strSize / 2, i+1) {
            if (strs[i] != strs[strSize-i-1]) {
              return false;
            }
          }
          return true;
        }
        main: function void() {
            strs   : array [5] of integer = {1, 2, 3, 4 ,5};

            if(isPalindrome(strs, 5)) printString("Correct!!!");

        }
